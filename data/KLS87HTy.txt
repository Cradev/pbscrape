int c=0,s=0,b=0; // samo crna, siva i bijela za primjer, on ce brojati koliko se koji puta unio
int cRank=0, sRank=0, bRank=0; // redoslijed unosa broja
int brojac;
for(brojac=0;;brojac++){
	scanf("%c", &amp;a);
	if(a=='c' || a=='C'){
		c++;
		if(cRank!=0) cRank=brojac;
	}
if(a=='s' || a=='S'){
		s++;
		if(sRank!=0) sRank=brojac;
	}
if(a=='b' || a=='B'){
		b++;
		if(bRank!=0) bRank=brojac;
	}
		

	if(a=='K' || a=='k') break;
}</p></body></html>
