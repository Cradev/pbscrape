<html><body><p>class Vehicule(object):

    def __init__(self, wheels, niceness):
        self.wheels = wheels
        self.niceness = niceness

    def __add__(self, other):
        num_wheels = self.wheels + other.wheels
        niceness = 'meh' if num_wheels &lt; 4 else 'nice'
        return Vehicule(num_wheels, niceness)

    def __str__(self):
        return 'Vehicule with ' + str(self.wheels) + ' wheels... ' + str(self.niceness)


if __name__ == '__main__':

    motorbike = Vehicule(2, 'meh')
    bike = Vehicule(2, 'meh')

    # where the fun happens
    new_vehicule = motorbike + bike

    print new_vehicule
