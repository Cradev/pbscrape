<html><body><p>def make_array_flat_again(array, result_arr=[])
  return unless array.is_a?(Array)

  array.each do |a|
    if a.is_a?(Array)
      make_array_flat_again(a, result_arr)
    else
      result_arr &lt;&lt; a
    end
  end
  
  result_arr
end</p></body></html>
