<html><body><p>#!/usr/bin/env bash

(set -x; brew update;)

(set -x; brew cleanup;)
(set -x; brew cask cleanup;)

red=`tput setaf 1`
green=`tput setaf 2`
reset=`tput sgr0`

casks=( $(brew cask list) )

for cask in ${casks[@]}
do
    versions=$(brew cask list $cask --versions)
    info=( $(brew cask info $cask) )
    latest=${info[1]}

    if [[ $versions =~ $latest ]]; then
        echo "${red}${cask}${reset} is ${green}up-to-date${reset}."
    else
        echo "${red}${cask}${reset} requires ${red}update${reset}."
        (set -x; brew cask install $cask --force;)
    fi
done</p></body></html>
