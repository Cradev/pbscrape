<html><body><p>using UnityEngine;
using System.Collections;
using UnityEngine.UI;
using System.Collections.Generic;

public class MoveScript : MonoBehaviour 
{
	public bool slaying = true;
	public int spiderDamage;
	public int playerDamage;
	public int totalSpiderDamage;
	public int totalPlayerDamage;
	public Text eventHappen;
	public string placeholder;
	public string placeholderGameOver;
	List<string> messages = new List<string> ();

	public void Start () {

		TurnTicker();
		StartCoroutine ("timeDelay");

	}
	IEnumerator timeDelay () {
		foreach (object o in messages) {
			eventHappen.text += o.ToString ();
			yield return new WaitForSeconds (1);
			
		}
	}

	public int RandNumGen () {
		return Random.Range (0, 11);

	}

	public void TurnTicker () {


		while (slaying == true) {
			

			spiderDamage = RandNumGen();
			playerDamage = RandNumGen();
			Debug.Log ("Player damage: " + totalPlayerDamage);
			Debug.Log ("Spider damage: " + totalSpiderDamage);

			if (totalPlayerDamage &gt;= 20) {

				placeholderGameOver = "You Win the fight!";
				Debug.Log ("win");

			} else if (totalSpiderDamage &gt;= 20) {

				placeholderGameOver = "You Lose the fight!";
				Debug.Log ("lose");
			} 
			if (totalSpiderDamage &gt;= 20 || totalPlayerDamage &gt;= 20) {
				slaying = false;
			}
			if (spiderDamage &lt; playerDamage) {
				totalPlayerDamage += playerDamage - spiderDamage;
				placeholder = ("You deal " + (playerDamage - spiderDamage) + " damage!" + System.Environment.NewLine).ToString();

			} else if (spiderDamage &gt; playerDamage) {
				totalSpiderDamage += spiderDamage - playerDamage;
				placeholder = ("The spider deals " + (spiderDamage - playerDamage) + " damage!" + System.Environment.NewLine).ToString();

			} else {

				placeholder = ("You deliver equal blows, 0 damage!" + System.Environment.NewLine).ToString();

			}
			messages.Add (placeholder);
			messages.Add (placeholderGameOver);


		}
	}
}</string></string></p></body></html>
