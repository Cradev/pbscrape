int countSumOfOddDigits(int, int, int);
int countSumOfEvenDigits(int, int, int); 

int main() {
    printf("Enter the number of digits in numbers: ");
    int numberOfDigits;
    scanf("%d", &amp;numberOfDigits);
   
    printf("Enter %d elements of the array: \n", SIZE_OF_ARRAY);
    int array[SIZE_OF_ARRAY];
    for(int i = 0; i &lt; SIZE_OF_ARRAY; i++) {
        scanf("%d", &amp;array[i]);
    }
   
    int numberOfHappyTickets = 0;
    int number;
    int sumOfEvenDigits;
    int sumOfOddDigits;
    for (int i = 0; i &lt; SIZE_OF_ARRAY; i++) {
        number = array[i];
        sumOfEvenDigits = countSumOfEvenDigits(number, 0, 1);
        sumOfOddDigits = countSumOfOddDigits(number, 0, 1);
     
        if(sumOfEvenDigits == sumOfOddDigits)
        {
            numberOfHappyTickets++;
        }
    }
   
    printf("Number of \"happy\" tickets: %d\n", numberOfHappyTickets);
    return 0;
}
 
int countSumOfOddDigits(int number, int sumOfOddDigits, int counter) {
    if (number != 0) {
        if (counter % 2 != 0) {
            int digit = number % 10;
            sumOfOddDigits += digit;
        }
        counter++;
        number /= 10;
        countSumOfOddDigits(number, sumOfOddDigits, counter);
    } else {
        return sumOfOddDigits;
    }
}
 
int countSumOfEvenDigits(int number, int sumOfEvenDigits, int counter) {
    if (number != 0) {
        if (counter % 2 == 0) {
            int digit = number % 10;
            sumOfEvenDigits += digit;
        }
        counter++;
        number /= 10;
        countSumOfEvenDigits(number, sumOfEvenDigits, counter);
    } else {
        return sumOfEvenDigits;
    }
