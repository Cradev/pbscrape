void MainWindow::on_pushButton_2_clicked()
{
        float a = 0.2;
        int b = 1.8;
        float x;
        float y;
        int n = 80;
        double dx = (b - a) / n;
        double integral = (sqrt(a+1)*cos(a*a) + sqrt(b+1)*cos(b*b))/2;
        for (int i = 1; i &lt; n; i++) {
        x = a + i*dx;
        y = sqrt(x+1)*cos(x*x);
        integral += y;
        }
        integral *= dx;
        QMessageBox::information (this, "Integral", QString("Result: %1").arg(integral));
}
