const
   BorderEntryInt = 2147483647;

var
   a, b, c, code: integer;
   St: string;
   Discriminant: longint;

function InputCheck(): integer;
begin
   while not ((code = 0) and (result &lt;&gt; 0) and (result &lt; 32767) and (result &gt; -32768)) do
   begin
      readln(St);
      val(St, result, code);
      if (code &lt;&gt; 0) or (result = 0) or (result &gt; 32767) or (result &lt; -32768)  then
         writeln('&#1055;&#1077;&#1088;&#1077;&#1084;&#1077;&#1085;&#1085;&#1072;&#1103; &#1085;&#1077; &#1076;&#1086;&#1083;&#1078;&#1085;&#1072; &#1073;&#1099;&#1090;&#1100; &#1088;&#1072;&#1074;&#1085;&#1086;&#1081; &#1085;&#1091;&#1083;&#1102; &#1080; &#1074;&#1099;&#1093;&#1086;&#1076;&#1080;&#1090;&#1100; &#1079;&#1072; &#1088;&#1072;&#1084;&#1082;&#1080; &#1076;&#1072;&#1085;&#1085;&#1099;&#1093; &#1091;&#1089;&#1083;&#1086;&#1074;&#1080;&#1081; ! &#1042;&#1074;&#1077;&#1076;&#1080;&#1090;&#1077; &#1079;&#1085;&#1072;&#1095;&#1077;&#1085;&#1080;&#1077; &#1087;&#1077;&#1088;&#1077;&#1084;&#1077;&#1085;&#1085;&#1086;&#1081; &#1087;&#1088;&#1072;&#1074;&#1080;&#1083;&#1100;&#1085;&#1086; !');
   end;
end;

begin
   writeln('&#1042;&#1074;&#1077;&#1076;&#1080;&#1090;&#1077; &#1079;&#1085;&#1072;&#1095;&#1077;&#1085;&#1080;&#1077; &#1087;&#1077;&#1088;&#1077;&#1084;&#1077;&#1085;&#1085;&#1086;&#1081; &#1072; (-32768 &lt; a &#1080; a &lt;&gt; 0 &#1080; a &gt; 32767)');
   a := InputCheck();
   writeln('&#1042;&#1074;&#1077;&#1076;&#1080;&#1090;&#1077; &#1079;&#1085;&#1072;&#1095;&#1077;&#1085;&#1080;&#1077; &#1087;&#1077;&#1088;&#1077;&#1084;&#1077;&#1085;&#1085;&#1086;&#1081; b (-32768 &lt; b &#1080; b &lt;&gt; 0 &#1080; b &gt; 32767)');
   b := InputCheck();
   writeln('&#1042;&#1074;&#1077;&#1076;&#1080;&#1090;&#1077; &#1079;&#1085;&#1072;&#1095;&#1077;&#1085;&#1080;&#1077; &#1087;&#1077;&#1088;&#1077;&#1084;&#1077;&#1085;&#1085;&#1086;&#1081; c (-32768 &lt; c &#1080; c &lt;&gt; 0 &#1080; c &gt; 32767)');
   c := InputCheck();
   Discriminant := b * b - 4 * a * c;
   case Discriminant of 
      0: writeln('&#1059;&#1088;&#1072;&#1074;&#1085;&#1077;&#1085;&#1080;&#1077; &#1080;&#1084;&#1077;&#1077;&#1090; &#1086;&#1076;&#1080;&#1085; &#1082;&#1086;&#1088;&#1077;&#1085;&#1100; = ', -b / (2 * a));
      1..BorderEntryInt: writeln('&#1055;&#1077;&#1088;&#1074;&#1099;&#1081; &#1082;&#1086;&#1088;&#1077;&#1085;&#1100; = ', (-b + sqrt(Discriminant)) / (2 * a), ' &#1042;&#1090;&#1086;&#1088;&#1086;&#1081; &#1082;&#1086;&#1088;&#1077;&#1085;&#1100; = ', (-b - sqrt(Discriminant)) / (2 * a))
   else
      writeln('&#1058;&#1072;&#1082; &#1082;&#1072;&#1082; &#1076;&#1080;&#1089;&#1082;&#1088;&#1080;&#1084;&#1080;&#1085;&#1072;&#1085;&#1090; &#1084;&#1077;&#1085;&#1100;&#1096;&#1077; &#1085;&#1091;&#1083;&#1103;, &#1090;&#1086; &#1091;&#1088;&#1072;&#1074;&#1085;&#1077;&#1085;&#1080;&#1077; &#1085;&#1077; &#1080;&#1084;&#1077;&#1077;&#1090; &#1076;&#1077;&#1081;&#1089;&#1090;&#1074;&#1080;&#1090;&#1077;&#1083;&#1100;&#1085;&#1099;&#1093; &#1088;&#1077;&#1096;&#1077;&#1085;&#1080;&#1081;')
   end;
end.</p></body></html>
