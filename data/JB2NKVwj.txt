        do
        {
                printf("Podaj pocz&#261;tkow&#261; mas&#281; substancji (m &gt;= 10)\noraz cotygodniowy procentowy ubytek masy (0%% - 100%%): ");

        }while(masa &lt; 10 || masa &gt; 100 || ubytek &lt;= 0 || ubytek &gt;= 100);
 
        printf("\n| Tydzie&#324; | masa | ubytek %% |\n|---------|------|----------|");
 
        for(int i = 1; masa &gt;= 1.0; i++)
        {      
                printf("\n|%8d |%6.2f|%9.2f |", i, masa, procent);
                masa=masa - masa * (ubytek/100);
				procent=procent - procent * (ubytek/100);
        }
        printf("\n-----------------------------");
        _getch();
        return 0;
