double start() {
	double czas;
	czas = clock();
	return czas;
}

double koniec(double start) {
	double czas;
	czas = clock() - start;
	return czas / CLOCKS_PER_SEC;
}

int main()
{
	double czas;
	int min = 20, max = -20;
	vector &lt; int &gt; tab;
	srand(time(NULL));
	czas = start();
	for (int i = 0; i &lt; 10000000; i++) {
		tab.push_back(rand() % 41 - 20);
		cout &lt;&lt; tab[i] &lt;&lt; " ";
		if (min &gt; tab[i])
			min = tab[i];
		if (max &lt; tab[i])
			max = tab[i];
	}
	czas = koniec(czas);
	cout &lt;&lt; endl;
	cout &lt;&lt; "Tworzenie tablicy zajelo " &lt;&lt; czas &lt;&lt; " sekund" &lt;&lt; endl;
	cout &lt;&lt; "Max=" &lt;&lt; max &lt;&lt; endl;
	cout &lt;&lt; "Min=" &lt;&lt; min &lt;&lt; endl;

	system("PAUSE");
