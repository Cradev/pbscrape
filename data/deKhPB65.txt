<html><body><p>class Solution {
public:
    bool isPalindrome(string s) {
	    int dlugosc = s.length();
        if (dlugosc == 1) 
        return true;
        int i=0; 
	    int z=dlugosc - 1;
        int a=0;
        while(a &lt;= z)
        {
            if ((s[a]&gt;64) &amp;&amp; (s[a]&lt;91)) 
            s[a]=s[a]+32; 
            a++;
        }
        while (i &lt; z) 
        {
            if ((s[i] &lt; 48) || (s[i] &gt; 57 &amp;&amp; s[i] &lt; 97) || (s[i] &gt; 122)) i++;
            else if ((s[z] &lt; 48) || (s[z] &gt; 57 &amp;&amp; s[z] &lt; 97) || (s[z] &gt; 122)) z--;
            else if (s[i++] != s[z--]) 
            return false;
        }
        return true;
    }
};</p></body></html>
