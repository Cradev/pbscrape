<html><body><p>#include <iostream>
#include <cmath>

using namespace std;

double f(double x, double y) {
	return tan(x) - y;
}

double solveEquation(double y, double eps) { // solves equation y = tg x; |answer - found_x| &lt; eps / 2
	double xLeft = 0, xRight = M_PI / 4;
	double fLeft = f(xLeft, y), fRight = f(xRight, y);
	while (xRight - xLeft &gt; eps) {
		double xMiddle = (xRight - xLeft) * fLeft / (fRight - fLeft);
		double fMiddle = f(xMiddle, y);
		if (fMiddle &lt; 0) {
			xLeft = xMiddle;
			fLeft = fMiddle;
		} else {
			xRight = xMiddle;
			fRight = fMiddle;
		}
	}
	return (xLeft + xRight) / 2;
}

int main() {
	double y = 0.57735;
	double x = solveEquation(y, 1e-8);
	cout &lt;&lt; "Radians: " &lt;&lt; x &lt;&lt; endl
		  &lt;&lt; "Degrees: " &lt;&lt; (x / M_PI * 180) &lt;&lt; endl
		  &lt;&lt; "tg(x): " &lt;&lt; tan(x) &lt;&lt; endl;
	return 0;
}</cmath></iostream></p></body></html>
