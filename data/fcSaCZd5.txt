#include <stdlib.h>
#include <math.h>

int main()
{
    float num;

    printf("Enter your natural number: ");
    scanf("%f", &amp;num);

    if(num &gt; 0) {
        int i, summa, a;
        float x, f;
        i = 1;
        a = 9;
        summa = 0;
        do {
           x = num / pow(10,i);
           if (x &lt; 1) {
                printf("There is %d characters in your number.\n", i);
                break;
           }
           i++;
        } while (x &gt; 0);
        while (i &gt; 0) {
           f = num / (a * pow(10,(i - 1)));
           if (f &gt;= 1) {
                summa = summa + a;
                num = num - (a * pow(10,(i - 1)));
                a = 10;
                i--;
           }
           if (num == 0)
                break;
           a--;
        }
        printf("Sum: %d", summa);
    }
    else
        printf("Error. Try again later.\n");
    return 0;
}</math.h></stdlib.h></stdio.h></p></body></html>
