import stanford.karel.*;

public class CheckerboardKarel extends SuperKarel {

	public void run() {
		fillTheRow();
		comeToNextRow();
	}
	/*
	 * precondition:Karel stands at the start point ,facing east
	 * postcondition:Karel stands at the end of the row,facing east
	 */

	private void fillTheRow() {
		if (frontIsClear()) {
			putBeeper();
			move();
		} else {
			turnLeftAndFillTheWallAbove();
		}

		while (frontIsClear()) {
			if (frontIsClear()) {
				move();
			}
			if (frontIsClear()) {
				putBeeper();
				move();
			} else {
				putBeeper();
			}
		}
	}

	/*
	 * precondition:Karel stands at the end of the first row 
	 * postcondition:Karel finishes the action from this method.Karel comes from one row to another
	 * and fills it. once from left and once from right
	 */
	private void comeToNextRow() {
		while (leftIsClear()) {
			if (beepersPresent()) {
				turnLeft();
				move();
				turnLeft();
				move();
				fillTheRow();
			} else {
				turnLeft();
				move();
				turnLeft();
				fillTheRow();
			}
			if (rightIsClear()) {
				if (beepersPresent()) {
					turnRight();
					move();
					turnRight();
					move();
					fillTheRow();
				} else {
					turnRight();
					move();
					turnRight();
					fillTheRow();
				}
			} else {
				turnAround();
			}
		}
	}

	/*
	 * precondition:Karel stands at the start point
	 * postcondition:Karel turns left and fills the
	 * wall above while front is clear.
	 */
	private void turnLeftAndFillTheWallAbove() {
		putBeeper();
		turnLeft();
		if (frontIsClear()) {
			move();
			move();
			putBeeper();
			move();
		}
	}
}</p></body></html>
