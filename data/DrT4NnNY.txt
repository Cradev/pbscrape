<html><body><p>public class GenericListEnumerator<x> : IEnumerator<x> where X : IComparable<x>
    {
        private GenericList<x> _collection = new GenericList<x>();
        private int index;
        private X _current;

        public GenericListEnumerator()
        {

        }

        public GenericListEnumerator(GenericList<x> collection)
        {
            _collection = collection;
            index = -1;
            _current = default(X);
        }



        public X Current
        {
            get
            {
                return _current;
            }
        }


        object IEnumerator.Current
        {
            get
            {
                return _current;
            }
        }
        public void Dispose()
        {
            _collection = null;
            _current = default(X);
            index = -1;
        }

        public bool MoveNext()
        {
            if (++index &gt;= _collection.Count)
            {
                return false;
            }
            else
            {
                _current = _collection.GetElement(index);
            }
            return true;
        }

        public void Reset()
        {
            _current = default(X);
            index = -1;
        }
    }
