import java.math.BigInteger;


/**
 * Created by Oskar on 21-Oct-16.
 */
public class Kalkulator {

    private static BigInteger obliczSilnie(int x) {
        BigInteger v = new BigInteger("" + 1);
        for (int i = 2; i &lt;= x; i++) {
            v = v.multiply(new BigInteger("" + i));
        }
        return v;// number formal exception
    }

    private static void zbadajDzialanie(String[] args){

        if(args.length == 1){
            int silniaInt = Integer.parseInt(args[0]);
            BigInteger silnia = obliczSilnie(silniaInt);
            System.out.println(silnia);
            return;
        }

        BigInteger liczba1Big = new BigInteger(args[0]);
        BigInteger liczba2Big = new BigInteger(args[2]);
        String rodzajDzialania = args[1];

        switch(rodzajDzialania){
            case "+":
                System.out.println("" + liczba1Big.add(liczba2Big));
                break;
            case "-":
                System.out.println("" + liczba1Big.subtract(liczba2Big));
                break;
            case "x":
                System.out.println("" + liczba1Big.multiply(liczba2Big));
                break;
            case "/":
                System.out.println("" + liczba1Big.divide(liczba2Big));
                break;
            case "^":
                int liczba2 = Integer.parseInt(args[2]);
                System.out.println("" + liczba1Big.pow(liczba2));
                break;
            case "mod":
                System.out.println("" + liczba1Big.mod(liczba2Big));
                break;
            default:
                System.out.println("brak operacji");
                break;
        }
    }

    public static void main(String[] args) {

        try {
            zbadajDzialanie(args);
        } catch (NumberFormatException e) {
            System.out.println("Nieprawidlowa liczba");
        }
    }
}</p></body></html>
