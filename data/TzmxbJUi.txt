# don't forget to make it executable : chmod u+x vpn.sh
# run it : ./vpn.sh $vpnName
if [ -z ${1} ]
 then
   echo "No VPN name given.";
   exit 1;
fi

while [[ true ]]; do
sleep 3;
nmcli connection show --active|grep vpn &gt; /dev/null;
if [ $? -ne 0 ]
 then
  nmcli connection up "$1";
fi
done</p></body></html>
