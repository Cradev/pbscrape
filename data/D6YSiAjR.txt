-export([flatten/1]).

%% @doc Flatten an array of arbitrarily nested arrays of integers into a flat 
%% array of integers. e.g. [[1,2,[3]],4] -&gt; [1,2,3,4]
-spec flatten(list())
	-&gt; list().

flatten([Head|Tail]) when is_list(Head) -&gt;
  lists:append(flatten(Head),flatten(Tail))
;

flatten([Head|Tail]) -&gt;
  lists:append([Head],flatten(Tail))
;


flatten(Item) -&gt;
	Item
