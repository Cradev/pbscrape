		DLLNode<e> ins = new DLLNode<e>(o,null,first);
		
		if (first == null)
			last = ins;
		else
			first.pred = ins;
		first = ins;
		
	}
	
	public void insertLast(E o) {
		
		if (first == null)

		}
	}
