public class NitricAcid extends BlockFluidClassic {

	protected IIcon nitricAcidStill;
	protected IIcon nitricAcidFlow;
	
	public NitricAcid(int id, Fluid fluid, Material material) {
		super(fluid, material);
		setCreativeTab(Generic.tabPootisFluids);
		setBlockName("nitricAcid");
	}


    public void registerIcons(IIconRegister icon)
    {
		this.nitricAcidStill = icon.registerIcon("pootis:nitricacidstill");
		this.nitricAcidFlow = icon.registerIcon("pootis:nitricacidflow");
    }
	
    
	@Override
    @SideOnly(Side.CLIENT)
	public IIcon getIcon(int side, int meta)
    {
		return side != 0 &amp;&amp; side != 1 ? this.nitricAcidFlow : this.nitricAcidStill;
