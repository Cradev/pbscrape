                aux = pontos[i];
                pontos[i] = pontos[j];
                pontos[j] = aux;

                strcpy(auxNome, nomes[i]);
                strcpy(nomes[i], nomes[j]);
                strcpy(nomes[j], auxNome);
            }
        }
    }
    // escrever de volta no txt

    arq = fopen(url, "w");

   if(arq == NULL){
        printf("ERRO: NAO FOI POSSIVEL ABRIR O ARQUIVO\n");
    }
    else{
        for(i = 0 ; i &lt; cont ; i++)
        fprintf(arq, "%d %s\n", pontos[i], nomes[i]);
    }
